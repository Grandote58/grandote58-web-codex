function dibujarPlano() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    ctx.strokeStyle = "#000";
    ctx.lineWidth = 2;

    // Eje X
    ctx.beginPath();
    ctx.moveTo(40, canvas.height - 40);

    // Eje Y

    ctx.moveTo(40, canvas.height - 20);

    // Flechas
    ctx.beginPath();
    ctx.moveTo(canvas.width - 25, canvas.height - 45);

// =============================
